use Namshi\JOSE\SimpleJWS;

$jws  = new SimpleJWS(array(
    'alg' => 'RS256'
));
$jws->setPayload(array(
    'id' => 42,
    'name' => 'John Doe',
    'exp' => time() + 3600
));

$privateKey = 'Clé privée RSA';
$jws->sign($privateKey);

echo $jws->getTokenString();
